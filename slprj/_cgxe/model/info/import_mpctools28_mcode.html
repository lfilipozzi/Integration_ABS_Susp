1function varargout = import_mpctools(local)
 2% `mpctools = import_mpctools()`
 3% `import_mpctools('*')`
 4%
 5% Returns a "module" struct with mpctools functions. This is similar to the
 6% Python construct `import mpctools`
 7%
 8% If the second form is used, the function handles are also created in the
 9% calling namespace. This is similar to the Python construct
10% `from mpctools import *`. Note that we do not advise using this form, as it
11% is only for lazy people who like namespace pollution.
12narginchk(0, 1);
13if nargin() > 0 && isequal(local, '*')
14    importstar = true();
15    nargoutchk(0, 0);
16elseif nargin() == 0
17    importstar = false();
18    nargoutchk(0, 1);
19else
20    error('Unknown argument: %s.', disp(local));
21end
22importlist = { ...
23    'c2d';
24    'collocweights';
25    'ekf';
26    'getCasadiFunc';
27    'getCasadiIntegrator';
28    'getCasadiDAE';
29    'getLinearizedModel';
30    'KalmanFilter';
31    'rk4';
32    'mpcplot';
33    'nlfilter';
34    'nmpc';
35    'nmhe';
36    'smushcolloc';
37    'spdinv';
38    'sstarg';
39    'jacobian';
40    'version';
41};
42
43module = struct();
44for i = 1:length(importlist)
45    func = importlist{i};
46    try
47        module.(func) = str2func(['mpctools.', func]);
48    catch err
49        error('Error importing %s: %s', func, err.message);
50    end
51    if importstar
52        assignin('caller', func, module.(func));
53    end
54end
55
56if importstar
57    varargout = {};
58else
59    varargout = {module};
60end
61
62end%function
