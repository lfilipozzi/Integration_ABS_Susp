1classdef RMPC < matlab.System & ...
  2        matlab.system.mixin.Propagates & ...
  3        matlab.system.mixin.SampleTime
  4%MPC Provide the MATLAB system to use MPC from the MPCTools toolbox in 
  5% Simulink. This requires to use interpreted execution. 
  6%   
  7%   The function requires the state of the model as an input and a structure
  8%   which define the parameters used in the model. The equations of motion
  9%   of the model have to be written in a separate function 
 10%   x_dot = MPC_EOM(x,u,param).
 11    
 12    % Public, tunable properties
 13    properties
 14        % Sampling time
 15        Delta = 0.01;
 16        % Number of state
 17        Nx = 5;
 18        % Number of input
 19        Nu = 3;
 20        % Number of disturbance
 21        Nw = 1;
 22        % Number of reference signals to follow
 23        Nr = 2;
 24        % Prediction horizon
 25        Nt = 20;
 26        % Slip target
 27        sx_ref = -0.3
 28    end
 29
 30    properties(Nontunable)
 31        % Structure array of parameters
 32        param = struct();
 33    end
 34
 35    properties(DiscreteState)
 36        % System state at the last timestep
 37        xprev
 38        % Control input at the last timestep
 39        uprev
 40        % Road preview
 41        roadprev
 42    end

 62    methods(Access = protected)
 63        %% setupImpl & stepImpl & resetImpl
 64        function setupImpl(obj)
 65            % Perform one-time calculations, such as computing constants
 66            
 67            % Compute matrices A, B, P, Q, R, K for slip control and torque
 68            % control mode.
 69            
 70            % +---------------------------+
 71            % |    Linearization point    |
 72            % +---------------------------+
 73            % Find steady-state condition for linearization
 74            % For torque control (ABS disengaged)
 75            [xss_TC, uss_TC] = steadyState(0,obj.param);
 76            % For slip control (ABS engaged)
 77            [xss_SC, uss_SC] = steadyState(-0.25,obj.param);
 78            
 79            % +---------------------------+
 80            % | Linearize and discretize  |
 81            % +---------------------------+
 82            [obj.A_TC,obj.B_TC, obj.Bd_TC] = ...
 83                obj.getLinearizedModel(xss_TC, uss_TC);
 84            [obj.A_SC,obj.B_SC, obj.Bd_SC] = ...
 85                obj.getLinearizedModel(xss_SC, uss_SC);
 86            
 87            % Extend model
 88            C = [0 0 0 0 1;
 89                 0 0 0 0 0;
 90                 1 0 0 0 0;
 91                 0 1 0 0 0];
 92            D = [0 0 0;
 93                 0 1 0;
 94                 0 0 0;
 95                 0 0 0];
 96            obj.A_TC = [obj.A_TC zeros(obj.Nx, obj.Nr);
 97                 C          eye(obj.Nr)];
 98            obj.B_TC = [obj.B_TC; D];
 99            
100            obj.A_SC = [obj.A_SC zeros(obj.Nx, obj.Nr);
101                 C          eye(obj.Nr)];
102            obj.B_SC = [obj.B_SC; D];
103            
104            % +---------------------------+
105            % |       Cost function       |
106            % +---------------------------+
107            % Stage cost and terminal cost
108            sx_norm   = 0.1;
109            tau_norm  = 1000;
110            Fc_norm   = 1000;
111            puns_norm = 50;
112            psp_norm  = 50;
113            
114            % Weight on states
115            qpuns = 50;     % Weight on puns
116            qpsp  = 50;     % Weight on qpsp
117            % Weights for torque control
118            q1_TC = 0;      % Tracking of slip
119            q2_TC = 1000;   % Tracking of torque
120            % Weights for slip control
121            q1_SC = 1000;   % Tracking of slip
122            q2_SC = 0;      % Tracking of torque
123            % Weight on actuator
124            qFc  = 1;   % Weight on rate of change of Fc
125            qtau = 100;     % Weight on rate of change of tau
126            
127            % Torque control
128            obj.Q_TC  = diag([...
129                0,...   % Penalty on rate of change of sprung mass momentum
130                0,...   % Penalty on rate of change of unsprung mass momentum
131                0,...   % Penalty on rate of change of suspension displacement
132                0,...   % Penalty on rate of change of tire displacmeent
133                0,...   % Penalty on rate of change of slip ratio
134                q1_TC/sx_norm^2,...     % Penalty on slip tracking error
135                q2_TC/tau_norm^2,...    % Penalty on torque target error
136                qpsp/psp_norm^2,...     % Penalty on sprung mass momentum
137                0]);                    % Penalty on unsprung mass momentum
138            
139            obj.R_TC  = diag([qtau/tau_norm^2 qFc/Fc_norm^2]);
140            
141            obj.P_TC = zeros(obj.Nx + obj.Nr);
142            obj.P_TC(1:5,1:5) = dare(obj.A_TC(1:5,1:5),obj.B_TC(1:5,1:2),...
143                obj.Q_TC(1:5,1:5),obj.R_TC);
144            
145            % Slip control
146            obj.Q_SC  = diag([...
147                0,...   % Penalty on rate of change of sprung mass momentum
148                0,...   % Penalty on rate of change of unsprung mass momentum
149                0,...   % Penalty on rate of change of suspension displacement
150                0,...   % Penalty on rate of change of tire displacmeent
151                0,...   % Penalty on rate of change of slip ratio
152                q1_SC/sx_norm^2,...     % Penalty on slip tracking error
153                q2_SC/tau_norm^2,...    % Penalty on torque target error
154                0,...                   % Penalty on the sprung mass momentum
155                qpuns/puns_norm^2]);    % Penalty on the unsprung mass momentum
156            
157            obj.R_SC  = diag([qtau/tau_norm^2 qFc/Fc_norm^2]);
158            
159            obj.P_SC = zeros(obj.Nx + obj.Nr);
160            obj.P_SC(1:5,1:5) = dare(obj.A_SC(1:5,1:5),obj.B_SC(1:5,1:2),...
161                obj.Q_SC(1:5,1:5),obj.R_SC);
162            
163            % +---------------------------+
164            % |    Controller of RMPC     |
165            % +---------------------------+
166            obj.K_TC = zeros(obj.Nu,obj.Nx);
167            obj.K_TC = zeros(obj.Nu,obj.Nx);
168            
169        end

171        function [u, solveTime] = stepImpl(obj, tau_ref, ...
172                road_prev, x, ABS_flag)
173            % Implement algorithm. 
174            
175            % Setup chronometer
176            tic;
177            time0 = toc;
178            
179            if ABS_flag == 1    % ABS is not engaged (torque control)
180                u = solveMPC(obj, obj.solver_TC, road_prev, x, tau_ref);
181            else                % ABS is engaged (slip control)
182                u = solveMPC(obj, obj.solver_SC, road_prev, x, tau_ref);
183            end
184            
185            % Record time
186            time1 = toc;
187            solveTime = time1 - time0;
188            
189        end

191        function resetImpl(obj)
192            % Initialize / reset discrete-state properties
193            obj.xprev = zeros(obj.Nx,1);
194            obj.uprev = zeros(obj.Nu,1);
195            obj.roadprev = zeros(1, obj.Nt);
196        end

199        function [A_disc,B_disc, Bd_disc] = getLinearizedModel(obj, xss, ~)
200            % Linearize the model around the operating point (xss, uss)
201            
202            % Load parameters
203            ksi = obj.param.ks;
204            bsi = obj.param.bs;
205            mi  = obj.param.msp;
206            kt  = obj.param.kt;
207            mus = obj.param.mus;
208            rw  = obj.param.rw;
209            B   = obj.param.B;
210            C   = obj.param.C;
211            D   = obj.param.D;
212            Jw  = obj.param.Jw;
213            g   = obj.param.g;
214            
215            % Linearization point
216            sxi = xss(5);
217            qti = xss(4);
218            
219            % Velocity treated as parameter in slip equation
220            U = 10/3.6;
221            
222            % Write continuous-time state-space
223            A_lin = [ -bsi/mi,  bsi/mus,  ksi,                                                                     0,                                                                                                                                                                               0;
224                   bsi/mi, -bsi/mus, -ksi,                                                                    kt,                                                                                                                                                                               0;
225                    -1/mi,    1/mus,    0,                                                                     0,                                                                                                                                                                               0;
226                        0,   -1/mus,    0,                                                                     0,                                                                                                                                                                               0;
227                        0,        0,    0, -(D*kt*rw^2*sin(C*atan(B*sxi))*((Jw*(sxi + 1))/(mi*rw^2) + 1))/(Jw*U), -(rw*((D*Jw*sin(C*atan(B*sxi))*(g*(mi + mus) + kt*qti))/(mi*rw) + (B*C*D*rw*cos(C*atan(B*sxi))*((Jw*(sxi + 1))/(mi*rw^2) + 1)*(g*(mi + mus) + kt*qti))/(B^2*sxi^2 + 1)))/(Jw*U)];
228            
229            B_lin = [  1,         0, 0;
230                -1,         0, 0;
231                 0,         0, 0;
232                 0,         0, 1;
233                 0, rw/(Jw*U), 0];
234                 
235            Bd_lin = [0; 0; 0; 1; 0];
236            
237            % Discretize the model
238            sys_cont = ss(A_lin, [B_lin Bd_lin], [], []);
239            sys_disc = c2d(sys_cont, obj.Dela);
240            
241            A_disc  = sys_disc.A;
242            B_disc  = sys_disc.B(:, 1:size(B_lin,2));
243            Bd_disc = sys_disc.B(:, size(B_lin,2)+1:end);
244            
245        end

247        %% Build min-max MPC problem
248        function u0 = RMPCSolver_solve(obj, P, Q, R, A, B, Bd, K, x0, ...
249                umin, umax, wmin, wmax)
250            % Solve the min-max MPC problem:
251            % min_u max_w xN' P xN + sum_k xk' Q xk + uk' R uk
252            % subkect to:
253            %       x_k+1 = A x_k + B u_k + Bd w_k
254            %       u_k = K x_k + v_k
255            %       x_0 = x0
256            %       u_min <= u_k <= u_max
257            %       w_min <= w <= w_max
258            
259            % Define AK
260            AK =A + B*K;
261            
262            % Check size:
263            if not(prod(size(umax) == [obj.Nu, 1]))
264                error('Incorect size for uub')
265            end
266            if not(prod(size(umin) == [obj.Nu, 1]))
267                error('Incorect size for ulb')
268            end
269            if not(prod(size(wmax) == [obj.Nw, 1]))
270                error('Incorect size for wub')
271            end
272            if not(prod(size(wmin) == [obj.Nw, 1]))
273                error('Incorect size for wlb')
274            end
275
276            % Check min <= max
277            if sum(umin >= umax)
278                error('Unfeasible bounds on u')
279            end
280            if sum(wmin >= wmax)
281                error('Unfeasible bounds on w')
282            end
283            
284            % Extend matrices umin, umax to obtain uub and ulb where:
285            %       ulb <= u <= uub
286            % with:
287            %       u = [u0 ... uN-1]^T
288            uub = zeros(obj.Nu*obj.Nt,1);
289            for k = 1:obj.Nt
290                uub((k-1)*obj.Nu+1:k*obj.Nu,:) = umax;
291            end
292
293            ulb = zeros(obj.Nu*obj.Nt,1);
294            for k = 1:obj.Nt
295                ulb((k-1)*obj.Nu+1:k*obj.Nu,:) = umin;
296            end
297            
298            % Solve the problem by using a batch approch:
299            % Write the state equation over the prediction horizon:
300            %       x = Sxx x0 + Sxv v + Sxw w
301            % where:
302            %       x = [x0 ... xN]^T
303            %       v = [v0 ... vN-1]^T
304            %       w = [w0 ... wN-1]^T
305
306            Sxx = zeros(obj.Nx*(obj.Nt+1),obj.Nx);
307            Sxx(1:obj.Nx,:) = eye(obj.Nx);
308            for i = 1:obj.Nt
309                Sxx(i*obj.Nx+1:(i+1)*obj.Nx,:) = AK * Sxx((i-1)*obj.Nx+1:i*obj.Nx,:);
310            end
311
312            Sxv = zeros(obj.Nx*(obj.Nt+1),obj.Nu*obj.Nt);
313            % Sv(Nx*Nt+1:end,Nu*(Nt-1)+1:end) = B;
314            for i = flip(2:obj.Nt+1)
315                Sxv(obj.Nx*(i-1)+1:obj.Nx*i, obj.Nu*(i-2)+1:obj.Nu*(i-1)) = B;
316                for j = flip(1:i-2)
317                    Sxv((i-1)*obj.Nx+1:i*obj.Nx, (j-1)*obj.Nu+1:j*obj.Nu) = ...
318                        AK * Sxv((i-1)*obj.Nx+1:i*obj.Nx, j*obj.Nu+1:(j+1)*obj.Nu);
319                end
320            end
321
322            Sxw = zeros(obj.Nx*(obj.Nt+1),obj.Nw*obj.Nt);
323            % Sv(Nx*Nt+1:end,Nu*(Nt-1)+1:end) = B;
324            for i = flip(2:obj.Nt+1)
325                Sxw(obj.Nx*(i-1)+1:obj.Nx*i, obj.Nw*(i-2)+1:obj.Nw*(i-1)) = Bd;
326                for j = flip(1:i-2)
327                    Sxw((i-1)*obj.Nx+1:i*obj.Nx, (j-1)*obj.Nw+1:j*obj.Nw) = ...
328                        AK * Sxw((i-1)*obj.Nx+1:i*obj.Nx, j*obj.Nw+1:(j+1)*obj.Nw);
329                end
330            end
331
332            % Write the control input equation over the prediction horizon
333            %       u = Sux x0 + Suv v + Suw w
334            % where:
335            %       u = [u0 ... uN-1]^T
336            %       v = [v0 ... vN-1]^T
337            %       w = [w0 ... wN-1]^T
338
339            KK = zeros(obj.Nt*obj.Nu,obj.Nt*obj.Nx);
340            for k = 1:obj.Nt
341                KK((k-1)*obj.Nu+1:k*obj.Nu,(k-1)*obj.Nx+1:k*obj.Nx) = K;
342            end
343
344            Sux = KK * Sxx(1:obj.Nt*obj.Nx,:);
345
346            Suv = eye(obj.Nt*obj.Nu);
347            Suv(obj.Nu+1:end,1:obj.Nu) = Sux(1:(obj.Nt-1)*obj.Nu,:) * B;
348            for j = 1:obj.Nt-1
349                Suv(j*obj.Nu+1:end,(j-1)*obj.Nu+1:j*obj.Nu) = Suv(obj.Nu+1:(obj.Nt-j+1)*obj.Nu,1:obj.Nu);
350            end
351
352            Suw = zeros(obj.Nt*obj.Nu);
353            Suw(obj.Nu+1:end,1:obj.Nu) = Sux(1:(obj.Nt-1)*obj.Nu,:) * Bd;
354            for j = 1:obj.Nt-1
355                Suw(j*obj.Nu+1:end,(j-1)*obj.Nu+1:j*obj.Nu) = Suw(obj.Nu+1:(obj.Nt-j+1)*obj.Nu,1:obj.Nu);
356            end
357
358            % Write cost function over the prediction horizon:
359            %       J = x' QQ x + u' RR u
360            QQ = zeros(obj.Nx*(obj.Nt+1));
361            for k = 1:obj.Nt
362                QQ((k-1)*obj.Nx+1:k*obj.Nx,(k-1)*obj.Nx+1:k*obj.Nx) = Q;
363            end
364            QQ(obj.Nt*obj.Nx+1:(obj.Nt+1)*obj.Nx,obj.Nt*obj.Nx+1:(obj.Nt+1)*obj.Nx) = P;
365
366            RR = zeros(obj.Nu*obj.Nt);
367            for k = 1:obj.Nt
368                RR((k-1)*obj.Nu+1:k*obj.Nu,(k-1)*obj.Nu+1:k*obj.Nu) = R;
369            end
370
371            % Reformulate the cost function using matrices Hx, Hv, and Hw:
372            %       J = || Hx x + Hv v + Hw w ||_2^2
373            % From the matrices that have been computed, the cost can be written as:
374            %   J = [x0' v' w'] * HH * [x0; v; w]
375            % with HH a symmetric positive semi-definit matrix
376            % 
377            % HH = [(Sxx'*QQ*Sxx + Sux'*RR*Sux) (Sxx'*QQ*Sxv + Sux'*RR*Suv) (Sxx'*QQ*Sxw + Sux'*RR*Suw);
378            %       (Sxv'*QQ*Sxx + Suv'*RR*Sux) (Sxv'*QQ*Sxv + Suv'*RR*Suv) (Sxv'*QQ*Sxw + Suv'*RR*Suw);
379            %       (Sxw'*QQ*Sxx + Suw'*RR*Sux) (Sxw'*QQ*Sxv + Suw'*RR*Suv) (Sxw'*QQ*Sxw + Suw'*RR*Suw)];
380            % 
381            % Where:
382            %   HH = [ Hx'*Hx   Hx'*Hv   Hx'*Hw;
383            %          Hv'*Hx   Hv'*Hv   Hv'*Hw;
384            %          Hw'*Hx   Hw'*Hv   Hw'*Hw];
385
386            % HxtHx = Sxx'*QQ*Sxx + Sux'*RR*Sux;
387            % HxtHv = Sxx'*QQ*Sxv + Sux'*RR*Suv;
388            % HxtHw = Sxx'*QQ*Sxw + Sux'*RR*Suw;
389            HvtHv = Sxv'*QQ*Sxv + Suv'*RR*Suv;
390            % HvtHw = Sxv'*QQ*Sxw + Suv'*RR*Suw;
391            HwtHw = Sxw'*QQ*Sxw + Suw'*RR*Suw;
392            HvtHx = Sxv'*QQ*Sxx + Suv'*RR*Sux;
393            HwtHx = Sxw'*QQ*Sxx + Suw'*RR*Sux;
394            HwtHv = Sxw'*QQ*Sxv + Suw'*RR*Suv;
395            
396            % Compute all vertices of the disturbance set
397            % Create all possible combination of disturbances given the bounds on w
398            % over the prediction horizon. Possible combinations are:
399            %       w_1 = [wmin ... wmin wmin wmin]^T
400            %       w_2 = [wmin ... wmin wmin wmax]^T
401            %       w_3 = [wmin ... wmin wmax wmin]^T
402            %       w_4 = [wmin ... wmin wmax wmax]^T
403            %       ...
404            %       w_{2^(Nw*Nt)} = [wmax ... wmax]^T
405            % The number of combination is exponential with the prediction horizon and
406            % number of disturbances.
407            % 
408            % To do that we create w_min_max the vector of extreme values for w for one
409            % timestep. Then, w_min_max is a cell structure of extreme values for each
410            % timestep over the prediction horizon. Finally, w_vertices is the vector
411            % of all possible combinations.
412
413            w_min_max = [wmin wmax];
414            w_min_max_horizon = cell(obj.Nt,1);
415            for k = 1:obj.Nt
416                w_min_max_horizon{k} = w_min_max;
417            end
418            w_vertices = allcomb(w_min_max_horizon{:})';
419            
420            % Each row of w_vertices correspond to a possible combination of
421            % disturbance at one timestep of the prediction horizon.
422
423            % TODO: this only work for one disturbance for now (Nw = 1)
424            
425            % Transform constraint on x0 and u into constraints on x0 and v
426            % With the following bounds on u:
427            %       ulb <= u <= uub
428            % we have:
429            %       ulb <= Sux x0 + Suv v + Suw w <= uub
430            % or equivalently
431            %      -Sux x0 - Suv v <= - ulb + Suw w
432            %       Sux x0 + Suv v <=   uub - Suw w
433
434            % Constraints can now be written as:
435            %       F x0 + Gv <= m + M w
436            F = [-Sux; Sux];
437            G = [-Suv; Suv];
438            m = [-ulb; uub];
439            M = [Suw; -Suw];
440
441            % Constraints can be written as:
442            %       F x0 + Gv <= d
443            % with:
444            %       d = m + max_{w} M*w
445            d = m + max(M*w_vertices,[],2);
446
447            % TODO: for now only bounds on u are considered, need to add bounds on x
448            % and polytopic constraints ax x +au u <= b
449            
450            % Transform min-max optimization problem into mp-QP
451            % The min-max MPC problem is equivalent to the following QP problem:
452            %   min_{v, gamma} || Hx * x + Hv * v + Hw * 0 ||_2^2 + gamma
453            %   subject to:   F*v + G*v <= d
454            %                 gamma >= gama_min(w) for all w
455            % 
456            % with gamma_min(w) = || Hx * x + Hv * v + Hw * w ||_2^2 
457            %                   - || Hx * x + Hv * v + Hw * 0 ||_2^2
458            %                   = w' Hw'Hw w + 2*w'Hw'*(Hx x + Hv v)
459            %
460            % To obtain an equivalent problem in which the functional does not depends 
461            % on the state vector, we change variable
462            %       z = v + (Hv'*Hv)^-1 Hv'Hx x
463            % and the min max problem becomes
464            %   J = x0' Y x0 + min_{z, gamma} 1/2*z'*H*z + gamma
465            %   subject to: Gm z + gm*gamma <= Wm + Sm * x
466            %               Gc z <= Wc + Sc x
467            % with H = 2*Hv'*Hv
468
469            H = 2*HvtHv;
470
471            % After changing variable, the constraints become:
472            %   G z <= d + [G (Hv'Hv)^(-1) Hv'Hx - F] x
473            % or equavlaently:
474            %   Gc z <= Wc + Sc x
475            % and for all w vertex of W:
476            % 2 w' Hw'Hv z - gamma <= -w' Hw'Hw w + w' 2*[Hw'Hx - Hw'Hv (Hv'Hv)^(-1) Hv'Hx] x
477            % which can be written as 2^(Nw*Nt) inequalities as:
478            %   Gm z - gamma <= Wm + Sm x
479
480            Gc = G;
481            Wc = d;
482            Sc = G/HvtHv*HvtHx - F;
483
484            % Check Wm: size error
485            Gm = 2 * w_vertices' * HwtHv;
486            gm = -ones(2^(obj.Nw*obj.Nt),1);
487            Wm = zeros(2^(obj.Nw*obj.Nt),1);
488            for k = 1:2^(obj.Nu*obj.Nt)
489                Wm(k) = -w_vertices(:,k)' * HwtHw * w_vertices(:,k);
490            end
491            Sm = 2 * w_vertices' * (HwtHx - HwtHv / HvtHv * HvtHx);
492            
493            % Reformulate the problem as a quadratic problem and solve it
494
495            % Make sure H is symmetric due to computation error
496            H = (H + H')/2;
497
498            H_QP = [H zeros(obj.Nu*obj.Nt,1); zeros(1,obj.Nu*obj.Nt) 0];
499            f_QP = [zeros(obj.Nu*obj.Nt,1); 1];
500            A_QP = [Gm gm; Gc zeros(2*obj.Nt*obj.Nu,1)];    % TODO: 2*Nt*Nu since only bounds on u are considered
501            b_QP = [(Wm + Sm*x0); (Wc + Sc*x0)];
502            
503            % TODO: add equality constraint to add road preview
504            z_QP = quadprog(H_QP,f_QP,A_QP,b_QP);
505            z  = z_QP(1:obj.Nu*obj.Nt);     % Get rid of gamma
506            v  = z - HvtHv \ HvtHx * x0;    % Compute sequence v
507            v0 = v(1:obj.Nu);   % Select the first entry
508            u0 = K*x0 + v0;     % Compute u0
509        end

511        %% Solve MPC problem
512        function [u, cost] = solveMPC(obj, solver, road_prev, x, tau_ref)
513            % Solve the MPC Problem for a given solver
514            
515            % Compute the extended state for integral formulation
516            xaug = [x-obj.xprev;        % State difference
517                x(5)-obj.sx_ref;        % Slip target error
518                obj.uprev(2)-tau_ref;   % Torque target error
519                x(1);                   % Sprung mass momentum
520                x(2)];                  % Unsprung mass momentum
521            
522            % Save the value of the current state for the next timestep
523            obj.xprev = x;
524            
525            % Set bounds on actuators
526            [umin, umax] = getActuatorBounds(obj, x);
527            
528            % Set bound on disturbance (TODO: set it depending on U)
529            wmin = -0.1;
530            wmax = 0.1;
531            
532            % Feed the road preview information to the MPC
533%             for k = 1:obj.Nt
534%                 solver.fixvar('u', k, road_prev(:,k) - obj.roadprev(:,k), 3)
535%             end
536            % Save road preview for next timestep
537            obj.roadprev = road_prev;
538            
539            % Solve the MPC problem
540            du = RMPCSolver_solve(obj.P_TC, obj.Q_TC, obj.R_TC, ...
541                    obj.A_TC, obj.B_TC, obj.Bd_TC, obj.K_TC, xaug, ...
542                umin, umax, wmin, wmax);
543            
544            % Compute the new control input
545            u = obj.uprev + du;
546            
547            % Save the new control input for the next timestep
548            obj.uprev = u;
549            
550            % Save the cost
551            cost = solver.obj;
552        end

554        %% Define actuator bounds
555        function [umin, umax] = getActuatorBounds(obj, state)
556            
557            umax = inf(obj.Nu, 1);
558            umin = -umax;
559            
560            % Unpack parameters:
561            a1 = obj.param.a1;
562            a2 = obj.param.a2;
563            a3 = obj.param.a3;
564            a4 = obj.param.a4;
565            a5 = obj.param.a5;
566            b1 = obj.param.b1;
567            b2 = obj.param.b2;
568            b3 = obj.param.b3;
569            b4 = obj.param.b4;
570            b5 = obj.param.b5;
571            msp = obj.param.msp;
572            mus = obj.param.mus;
573            
574            % +-----------------------+
575            % |   Suspension bounds   |
576            % +-----------------------+
577            % Define bounds depending on the suspension relative velocity
578            pspr = state(1);
579            puns = state(2);
580            vrel = -(pspr/msp - puns/mus);
581            vm = (a1-a4) / (b4-b1);
582            vp = (a1-a3) / (b3-b1);
583            if vrel < vm
584                umin(1) = max(b1 * vrel + a1, b5 * vrel + a5);
585                umax(1) = (b4 * vrel + a4);
586            elseif vrel <= vp
587                umin(1) = (b1 * vrel + a1);
588                umax(1) = (b1 * vrel + a1);
589            else
590                umin(1) = (b3 * vrel + a3);
591                umax(1) = min(b1 * vrel + a1, b2 * vrel + a2);
592            end
593            
594            % +-----------------------+
595            % |     Brake  bounds     |
596            % +-----------------------+
597            % Braking torque must always be negative
598            umax(2) = 0;
599            
600            % +-----------------------+
601            % | Integral formulation  |
602            % +-----------------------+
603            % Translate the bounds to convert the constraint on u into a
604            % constraint on du = u(k) - u(k-1)
605            umin = umin - obj.uprev;
606            umax = umax - obj.uprev;
607        end
